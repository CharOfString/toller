38void print_level_prefix(log_level_t level) {
39 switch (level) {
40 case LOG_INFO:
41 fprintf(stdout, "%s", BOLD_BLUE_FORMAT);
42 fprintf(stdout, "[INFO]");
43 fprintf(stdout, "%s ", RESET_FORMAT);
44 break;
45 case LOG_OK:
46 fprintf(stdout, "%s", BOLD_GREEN_FORMAT);
47 fprintf(stdout, "[ OK ]");
48 fprintf(stdout, "%s ", RESET_FORMAT);
49 break;
50 case LOG_WARNING:
51 fprintf(stdout, "%s", BOLD_YELLOW_FORMAT);
52 fprintf(stdout, "[WARNING]");
53 fprintf(stdout, "%s ", RESET_FORMAT);
54 break;
55 case LOG_ERROR:
56 fprintf(stderr, "%s", BOLD_RED_FORMAT);
57 fprintf(stderr, "[ERROR]");
58 fprintf(stderr, "%s ", RESET_FORMAT);
59 break;
60 default:
61 fprintf(stdout, "%s", BOLD_CYAN_FORMAT);
62 fprintf(stdout, "[DEBUG] ");
63 fprintf(stdout, "%s ", RESET_FORMAT);
64 break;
65 }
66}

83void write_log(log_level_t level, const char* content, ...) {
84 print_level_prefix(level);
85 va_list args;
86 va_start(args, content);
87 vfprintf(level == LOG_ERROR ? stderr : stdout, content, args);
88 fprintf(level == LOG_ERROR ? stderr : stdout, "\n");
89 va_end(args);
90}
