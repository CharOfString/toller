33static char* duplicate_string(const char* src) {
34 size_t len;
35 char* dst;
36
37 if (src == NULL) {
38 return NULL;
39 }
40
41 len = strlen(src) + 1;
42 if (len > MAX_PATH_LENGTH) {
43 return NULL;
44 }
45
46 dst = malloc(len);
47 if (dst == NULL) {
48 return NULL;
49 }
50
51 memcpy(dst, src, len);
52 return dst;
53}

69path_query_response_t get_system_data_dir() {
70 path_query_response_t response;
71 const char* xdg_data = getenv("XDG_DATA_HOME");
72 if (xdg_data != NULL) {
73 response.path = duplicate_string(xdg_data);
74 response.status_code = (response.path != NULL) ? 0 : -2;
75 return response;
76 } else {
77 const char* home_dir = getenv("HOME");
78 if (home_dir != NULL) {
79 char temp_path[MAX_PATH_LENGTH];
80 int written = snprintf(temp_path, sizeof(temp_path),
81 "%s/.local/share", home_dir);
82
83 if (written < 0 || written >= (int) sizeof(temp_path)) {
84 // While exceeding the max buffer size
85 write_log_with_tag(LOG_ERROR, "Path Handler",
86 "Error getting path",
87 "The generated system data directory path exceeds the ",
88 "MAX_PATH_LEN");
89 response.status_code = -2;
90 response.path = NULL;
91 return response;
92 }
93
94 response.path = duplicate_string(temp_path);
95 if (response.path == NULL) {
96 response.status_code = -2;
97 return response;
98 }
99
100 response.status_code = -1;
101 return response;
102 } else {
103 response.status_code = -2;
104 response.path = NULL;
105 return response;
106 }
107 }
108}

124path_query_response_t get_application_data_dir() {
125 path_query_response_t response;
126
127 path_query_response_t system_data_dir = get_system_data_dir();
128 if (system_data_dir.status_code == -2) {
129 write_log_with_tag(LOG_ERROR, "Path Handler",
130 "Error getting path",
131 "The path you requested depends on the system data directory, ",
132 "but we failed to get that path.");
133 response.status_code = -2;
134 response.path = NULL;
135 return response;
136 }
137
138 struct stat data_dir_st;
139 if (!(stat(response.path, &data_dir_st) == 0 &&
140 S_ISDIR(data_dir_st.st_mode))) {
141 write_log_with_tag(LOG_ERROR, "Path Handler",
142 "Error getting path",
143 "The path you requested depends on the system data directory, ",
144 "but seems that it does NOT exist on your system.");
145 free(system_data_dir.path);
146 response.status_code = -2;
147 response.path = NULL;
148 return response;
149 }
150
151 char temp[MAX_PATH_LENGTH];
152 int written = snprintf(temp, sizeof(temp), "%s/%s",
153 system_data_dir.path, APPLICATION_NAME_LOWER);
154 free(system_data_dir.path);
155
156 if (written < 0 || written >= (int) sizeof(temp)) {
157 // While exceeding the max buffer size
158 write_log_with_tag(LOG_ERROR, "Path Handler",
159 "Error getting path",
160 "The generated application data directory path exceeds the ",
161 "MAX_PATH_LEN");
162 response.status_code = -2;
163 response.path = NULL;
164 return response;
165 }
166
167 struct stat app_data_dir_st;
168 if (!(stat(temp, &app_data_dir_st) == 0 &&
169 S_ISDIR(app_data_dir_st.st_mode))) {
170 if (mkdir(temp, 0755) != 0) {
171 write_log_with_tag(LOG_ERROR, "Path Handler",
172 "Error getting path",
173 "Failed to create the application data path.");
174 response.status_code = -2;
175 response.path = NULL;
176 free(temp);
177 return response;
178 }
179 }
180
181 response.path = duplicate_string(temp);
182 if (response.path == NULL) {
183 write_log_with_tag(LOG_ERROR, "Path Handler",
184 "Error getting path",
185 "Failed to copy the application data path to response.");
186 response.status_code = -2;
187 return response;
188 }
189
190 response.status_code = 0;
191 return response;
192}
