25static char* duplicate_string(const char* src) {
26 size_t len;
27 char* dst;
28
29 if (src == NULL) {
30 return NULL;
31 }
32
33 len = strlen(src) + 1;
34 if (len > MAX_PATH_LENGTH) {
35 return NULL;
36 }
37
38 dst = malloc(len);
39 if (dst == NULL) {
40 return NULL;
41 }
42
43 memcpy(dst, src, len);
44 return dst;
45}

60path_query_response_t get_system_data_dir() {
61 path_query_response_t response;
62 const char* xdg_data = getenv("XDG_DATA_HOME");
63 if (xdg_data != NULL) {
64 response.path = duplicate_string(xdg_data);
65 response.status_code = (response.path != NULL) ? 0 : -2;
66 return response;
67 } else {
68 const char* home_dir = getenv("HOME");
69 if (home_dir != NULL) {
70 char temp_path[MAX_PATH_LENGTH];
71 int written = snprintf(temp_path, sizeof(temp_path),
72 "%s/.local/share", home_dir);
73
74 if (written < 0 || written >= (int)sizeof(temp_path)) {
75 // While exceeding the max buffer size
76 write_log_with_tag(LOG_ERROR, "Path Handler",
77 "Error getting path",
78 "The generated system data directory path exceeds the ",
79 "MAX_PATH_LEN");
80 response.status_code = -2;
81 response.path = NULL;
82 return response;
83 }
84
85 response.path = duplicate_string(temp_path);
86 if (response.path == NULL) {
87 response.status_code = -2;
88 return response;
89 }
90
91 response.status_code = -1;
92 return response;
93 } else {
94 response.status_code = -2;
95 response.path = NULL;
96 return response;
97 }
98 }
